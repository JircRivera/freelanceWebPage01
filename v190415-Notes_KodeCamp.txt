# Tipos de selector
$("#").doSomething()                       // por ID
$(".").doSomething()                       // por Clase 
$("tagname").doSomething()                 // por tagname
$(".some-class:eq()").doSomething()        // por posicion
$("#some-id .some-class").doSomething()    // por anidacion
$("#some-id.some-class").doSomething()     // por conjuncion
$("#some-id:hover").doSomething()          // por estado

# Methods

$("input").fadeOut()                    // Desaparecer todos los elementos input
